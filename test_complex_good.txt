/* 1. Definim Clasa */
opera Compozitor {
    basso varsta;
    soprano talent;
    libretto nume;
}

/* 2. Variabile Globale */
Compozitor mozart;
basso ani;
soprano scor;

/* 3. Functie de calcul */
soprano calculeazaScor(basso v, soprano t) {
    basso experienta;
    experienta := v * 2;
    /* Returnam experienta + talent */
    /* Atentie: in implementarea ta simpla returneaza ultima valoare evaluata sau trebuie asignata */
    /* Simulam return-ul prin valoarea expresiei */
    experienta + t; 
}

/* 4. EXECUTION */
finale {
    /* Setam campurile obiectului (Lazy Instantiation la lucru!) */
    mozart.nume := "Wolfgang";
    mozart.varsta := 35;
    mozart.talent := 99.9;

    canta("Compozitorul este:");
    canta(mozart.nume);

    /* Folosim o bucla */
    ani := 0;
    canta("Se antreneaza...");
    durante (ani < 3) {
        mozart.talent += 10.5;
        ani++;
    }

    /* Apelam functia */
    scor := calculeazaScor(mozart.varsta, mozart.talent);
    
    canta("Scor Final:");
    canta(scor);

    se (scor > 100.0) {
        canta("Este un geniu!");
    }
}
