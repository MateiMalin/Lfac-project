/* TEST 6: Definire Clasa si Apel Metoda 
   Obiectiv: Verificam daca 'pavarotti.cantaAria()' functioneaza.
*/

/* 1. DEFINIM CLASA SI METODELE */
opera Solist {
    basso volum;
    
    /* Metoda care cere un String si un Int */
    silenzio cantaAria(libretto titlu, basso durata) {
        canta(titlu);
        canta(durata);
    }
}

/* 2. DECLARAM VARIABILE GLOBAL (Regula ta) */
Solist pavarotti;
libretto melodie;
basso timp;

/* 3. EXECUTIE */
finale {
    /* Initializare */
    melodie := "Nessun Dorma";
    timp := 5;

    /* Apelam metoda obiectului cu parametrii corecti */
    pavarotti.cantaAria(melodie, timp);
    
    /* Apelam cu literali */
    pavarotti.cantaAria("O sole mio", 3);
}
